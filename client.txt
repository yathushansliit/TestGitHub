App.jsx

import React, { Component } from 'react';
import Courses from './components/Courses';
import axios from "axios/index";

class App extends Component {
    constructor(){
        super();
        this.state = {
            courseItems:{}
        }
        this.getAllCourses();
    }
    getAllCourses(){
        let t = this;
        axios.get('http://localhost:8000/courses/').then(function (response) {
            let courses = response.data;
            t.setState({
                courseItems:courses
            });
        });
    }
  render() {
    return (
      <div>
          <h1>Following Courses Are Available </h1>
          <Courses courses={this.state.courseItems}/>
      </div>
    );
  }
}

export default App;
